Element

Both provided h and createElement provide the same interface to create virutal elements that Dyo understands.


h(
	type: Type
	props?: Props
	...children: [Children]
)

createElement(
	type: Type
	props?: Props
	...children: [Children]
)

Type

Dyo supports rendering many different types of element.


type Type = string|function|object

h(
	type: Type
)

createElement(
	type: Type
)

Host Element


type Host = string

h('h1')

createElement('h1')

Component Element


type Component = function

h(() => {
	return createElement('h1')
})

createElement(() => {
	return createElement('h1')
})

Async Element


type Async = Promise

h(Promise.resolve('Hello'))

createElement(Promise.resolve('Hello'))

Props


interface Props {
	[prop: string]: any
	on[event: string]: Handler
	key?: Key
	ref?: Ref
}

h(
	type: Type
	props?: Props
)

createElement(
	type: Type
	props?: Props
)

Keys


type Key = string|number|symbol

h(Type, {
	key: 'heading'
})

createElement(Type, {
	key: 'heading'
})

Refs


type Ref = string|function|object

string


h(Type, {
	ref: 'heading'
})

createElement(Type, {
	ref: 'heading'
})

function


h(Type, {
	ref: (value: object) => {}
})

createElement(Type, {
	ref: (value: object) => {}
})

object


h(Type, {
	ref: {current: value}
})

createElement(Type, {
	ref: {current: value}
})

Events


type Handler = function|EventListener|Array

interface EventListener {
	handleEvent(e) {}
}

Event Function


h('button', {
	onClick: (e) => {}
})

Event Listener


h('button', {
	onClick: {
		handleEvent: (e) => {}
	}
})

Event Array


h('button', {
	onClick: [(e) => {}, (e) => {}]
})

Children


type Children = function|string|number|element|iterable|thenable

h(
	type: Type
	props?: Props
	...children?: Children
)

createElement(
	type: Type
	props?: Props
	...children?: Children
)
