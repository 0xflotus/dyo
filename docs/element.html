Element

The createElement function provides an interface to create virtual elements that Dyo understands.


createElement(
	type: Type
	props?: Props
	...children: [Children]
)

The h function: pronounced "hyperscript" provides a short alias to the createElement interface.


h(
	type: Type
	props?: Props
	...children: [Children]
)

Type

Dyo accepts a variety of different element types.


type Type = string|function|object

h(
	type: Type
)

Host Element

An element is called a host element when the type is typeof "string".


type Host = string

h('h1')

Component Element

An element is called a component element when the type is typeof "function".


type Component = function

h(() => {
	return createElement('h1')
})

Exotic Element

An element is called a exotic element when the type is typeof "object". The most common type of exotic element is async elements: When the type is a thenable.


type Exotic = object

h(Promise.resolve('Hello'))

Props


interface Props {
	[prop: string]: any
	on[event: string]: Handler
	key?: Key
	ref?: Ref
}

h(
	type: Type
	props?: Props
)

Keys


type Key = string|number|symbol

h(Type, {
	key: 'heading'
})

Refs


type Ref = string|function|object

string


h(Type, {
	ref: 'heading'
})

function


h(Type, {
	ref: (value: object) => {}
})

object


h(Type, {
	ref: {current: value}
})

Events


type Handler = function|EventListener|Array<function|EventListener>

interface EventListener {
	handleEvent(e) {}
}

Event Function


h('button', {
	onClick: (e) => {}
})

Event Listener


h('button', {
	onClick: {
		handleEvent: (e) => {}
	}
})

Event Array


h('button', {
	onClick: [(e) => {}, (e) => {}]
})

Children


type Children = function|string|number|element|iterable|thenable

h(
	type: Type
	props?: Props
	...children?: Children
)
